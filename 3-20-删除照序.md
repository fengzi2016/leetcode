class Solution {
public:
    int minDeletionSize(vector<string>& A) {
        int r = A.size();
        if(r==0) return 0;
        int c = A[0].size();
        int res = 0;
          for(int j = 0; j < c; j++){
            for(int i = 0; i < r - 1; i++){
                if(A[i][j]>A[i+1][j]){
                  res++;
                  break;
                }
          }
        }
        return res;
    }
};
