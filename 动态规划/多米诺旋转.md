class Solution {
public:
    int minDominoRotations(vector<int>& A, vector<int>& B) {
        int count = 0;
        int same = 1;
        int j;
        for(int i = 0; i < A.size(); i++){
            for(j = 0; j < B.size(); j++){
                if(i == j) continue;
                if(A[i] == B[j] && A[i] != A[j]  ){
                    count++; 
                    same++;
                } else if( A[j] == A[i] && A[j] == B[j]) {
                    same++;
                } else if(A[i] == A[j] ){
                    same ++;
                }
            }
             if(same != B.size()) {
                count = 0;
                same = 1;
            }else {
                 break;
             }
        }
        
        if(j == B.size() && count == 0) return -1;
        return count;
    }
    
};
